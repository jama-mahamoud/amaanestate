// =================================================== // NO CACHE (PREVENT OLD PREVIEWS) // =================================================== res.setHeader("Cache-Control", "no-store, no-cache, must-revalidate"); res.setHeader("Pragma", "no-cache"); res.setHeader("Expires", "0"); return res.status(200).send(template); } catch (err) { console.error("[SEO ERROR]", err); return next(); } } // =================================================== // 4.
